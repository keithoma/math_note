\begin{defbox}
    \begin{definition}
        A \(K\)-algebra \(\mathcal{A}\) is a ring that is also a \(K\)-vector space, such that for \(\lambda \in K\) and \(a, b \in \mathcal{A}\) such that
        \begin{align}
            \lambda (ab) = (\lambda a) b = a (\lambda b)
        \end{align}
    \end{definition}
\end{defbox}

\begin{example}
    \begin{enumerate}
        \item ring containing \(K\) such as field extensions
        \item \(K[X, Y]\)
        \item \(K[[X]]\)
        \item \(\text{Mat}_{n\times n}(K)\)
    \end{enumerate}
\end{example}

\begin{defbox}
    \begin{definition}
        A \(R\)-algebra \(\mathcal{A}\) is a ring that is also an \(R\)-module.
    \end{definition}
\end{defbox}

\begin{theorem}
    Let \(K\) be a field that is finitely generated as a \(\mathbb{Z}\)-module then its finite.
\end{theorem}
\begin{proof}
    \begin{enumerate}
        \item Let \(x \in K\).
        \item \(x = z_1 x_1 + \cdots + z_n x_n\)
    \end{enumerate}
\end{proof}
