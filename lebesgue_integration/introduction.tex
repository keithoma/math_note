    \begin{align}
        g: [a, b] \rightarrow \mathbb{R}, x \mapsto g(x) :=
        \begin{cases}
            1 \text{ for } x \in \mathbb{Q},\\
            0 \text{ for } x \in \mathbb{R} \setminus \mathbb{Q} \text{.}
        \end{cases}
    \end{align}
\end{example}
